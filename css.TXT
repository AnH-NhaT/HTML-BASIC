-- float (left-right-none): canh lề trái phải
-- display (block): tạo khối bao cho phần tử
-- display (inline): cho các phần tử nằm trên một dòng
-- display (none): ẩn phần tử 100%.
-- display (inline-block): vừa inline vừa block
-- display (list-item): list xuống dòng
-- font-family: định dạng font chữ cho phần tử
-- font-size: cỡ chữ
-- font-style (italic-normal): kiểu chữ nghiêng-/-thường
-- font-weight (bold): độ đậm nhạt, dày mỏng của chữ
-- text-align (left-right-center-justify): căn lề cho chữ chiều ngang
-- vertical-align (top-middle-bottom): căn lề cho chữ chiều ngang
-- text-decoration-line (none-underline-): gạch chân, đường kẻ qua chữ
-- text-decoration-color: đặt màu cho đường kẻ (text-decoration-line)
-- text-decoration-style (solid-double): nét liền, nét đôi cho (text-decoration-line)
-- text-indent: (100px;): thụt lề
-- cursor: (pointer-wait): thiết lập hình dạng mouse khi gặp phải
-- possition (sticky): dính ở màn hình, không bị cuộn trôi
-- possition (static): vô hiệu hóa (top-right-bottom-left)
-- possition (relative): (top-right-bottom-left) với những cái gần cạnh nó
-- possition (fixed): (top-right-bottom-left) vị trí cố định từ cạnh màn hình,
 không chiếm chỗ trống trên trang, dù cuộn cũng không đổi vị trí
-- possition (sticky): (position: sticky;top: 27px; bottom: 31px;) thử là biết
-- padding : đệm thêm khoảng cách từ nội dung đến khung border (tăng size view)
-- margin : khoảng cách lề giữa các view
-- list-style: (outside none none;) bỏ dấu chấm đầu dòng thẻ li
